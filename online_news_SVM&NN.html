def shares_to_labels(a):
    """
    A function to read targer i.e. 
    share values and those below 1300 shares 
    are turned to class 0, and above 1300 are
    lablelled as class 1.
    """
    if -1<a<1300.0:
        l=0
    else:
        l=1
    return l
